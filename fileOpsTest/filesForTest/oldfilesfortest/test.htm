Example 1 - Reading Files

Go by Example Reading Files

ioutil

package iotuil

package main

import (

"bufio"

"fmt"

"io"

"io/ioutil"

"os"

)

 

// Reading files requires checking most calls for errors.

// This helper will streamline our error checks below.

func check(e error) {

if e != nil {

panic(e)

}

}

func main() {

 

// Perhaps the most basic file reading task is

// slurping a file's entire contents into memory.

dat, err := ioutil.ReadFile("/tmp/dat")

check(err)

fmt.Print(string(dat))

`os` Package Open, Seek and Read

os Package - Open

os Package - File.Seek

os Package - File.Read

// You'll often want more control over how and what

// parts of a file are read. For these tasks, start

// by `Open`ing a file to obtain an `os.File` value.

f, err := os.Open("/tmp/dat")

check(err)

 

// Read some bytes from the beginning of the file.

// Allow up to 5 to be read but also note how many

// actually were read.

b1 := make([]byte, 5)

n1, err := f.Read(b1)

check(err)

fmt.Printf("%d bytes: %s\n", n1, string(b1))

 

// You can also `Seek` to a known location in the file

// and `Read` from there.

o2, err := f.Seek(6, 0)

check(err)

b2 := make([]byte, 2)

n2, err := f.Read(b2)

check(err)

fmt.Printf("%d bytes @ %d: %s\n", n2, o2, string(b2))

`io` Package ReadAtLeast

Package io - ReadAtLeast

 

The `io` package provides some functions that may

// be helpful for file reading. For example, reads

// like the ones above can be more robustly

// implemented with `ReadAtLeast`.

o3, err := f.Seek(6, 0)

check(err)

b3 := make([]byte, 2)

n3, err := io.ReadAtLeast(f, b3, 2)

check(err)

fmt.Printf("%d bytes @ %d: %s\n", n3, o3, string(b3))

 

// There is no built-in rewind, but `Seek(0, 0)`

// accomplishes this.

_, err = f.Seek(0, 0)

check(err)

`bufio` Package NewReader and Peek

`bufio` Package - NewReader

`bufio` Package - Reader.Peek

// The `bufio` package implements a buffered

// reader that may be useful both for its efficiency

// with many small reads and because of the additional

// reading methods it provides.

r4 := bufio.NewReader(f)

b4, err := r4.Peek(5)

check(err)

fmt.Printf("5 bytes: %s\n", string(b4))

 

// Close the file when you're done (usually this would

// be scheduled immediately after `Open`ing with

// `defer`).

f.Close()

 

}

Bz2 NewReader Example

socketloop bz2 newreader example

package main

import (

"compress/bzip2"

"fmt"

"io"

"os"

)

 

func main() {

 

inputFile, err := os.Open("file.txt.bz2")

 

if err != nil {

fmt.Println(err)

os.Exit(1)

}

 

defer inputFile.Close()

outputFile, err := os.Create("file.txt")

if err != nil {

fmt.Println(err)

os.Exit(1)

}

 

defer outputFile.Close()

bzip2reader := bzip2.NewReader(inputFile)

 

if err != nil {

fmt.Println(err)

os.Exit(1)

}

 

io.Copy(outputFile, bzip2reader)

}

convert-cast-bytes-buffer-or-bytes-newbuffer-type-to-io-reader

You need to convert or type cast bytes.Buffer or bytes.NewBuffer type to io.Reader to use in io.MultiReader() function because of this error :

cannot use buffer_slice (type []*bytes.Buffer) as type io.Reader in argument to io.MultiReader: []*bytes.Buffer does not implement io.Reader (missing Read method).

socketloop tutorials

Solution : Wrap the bytes.Buffer with io.Reader array.

cast-bytes-buffer - 1

package main

import (

"bytes"

"fmt"

"io"

"io/ioutil"

)

 

func main() {

 

readerBuffer := bytes.NewBuffer([]byte("abcdefghijkl"))

 

readerBuffer2 := bytes.NewBuffer([]byte("mnopqrstuvwxyz"))

 

buff := []io.Reader{readerBuffer, readerBuffer2}

 

combined := io.MultiReader(buff...)

 

data, _ := ioutil.ReadAll(combined)

 

fmt.Println(string(data))

}

Read Text File Line by Line

Scanner Example

Source Code

Package bufio

This code reads a file line by line using bufio and 'scanner'. Note that scanner will strip off the new line character when returning the string from scanner.Text().

 

package main

import (

"bufio"

"fmt"

"os"

)

 

func main() {

 

inFile, err := os.Open("./TestText.txt")

 

defer inFile.Close()

 

if err != nil {

panic(err)

}

 

scanner := bufio.NewScanner(inFile)

 

scanner.Split(bufio.ScanLines)

 

for scanner.Scan() {

 

line := scanner.Text()

 

fmt.Printf("%s\n", line)

// Note the line received from scanner does

// NOT contain a new line.

}

 

}

Line # 2 Some Text

Line # 3 Some Text

Line # 4 Some Text

ReadString Example

Source Code

Package bufio

This example reads a text file line by line using bufio and Readstring. Note that the '/n' new line character is included in the returned string.

package main

import (

"bufio"

"fmt"

"io"

"os"

)

 

func checkError(e error, msg string) {

 

if e != nil {

fmt.Println(msg)

panic(e)

}

 

}

 

func main() {

 

f, err := os.Open("./TestText.txt")

 

checkError(err, "File failed to open!")

 

defer f.Close()

 

reader := bufio.NewReader(f)

 

for {

 

text, err := reader.ReadString('\n')

 

if err != nil && err == io.EOF {

break

}

 

checkError(err, "Read String Error")

 

fmt.Printf("%s", text)

 

}

}

Output

$ go run main.go

Line # 1 Some Text

Line # 2 Some Text

Line # 3 Some Text

Line # 4 Some Text
